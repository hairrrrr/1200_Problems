#### [78. 子集](https://leetcode-cn.com/problems/subsets/)



```
给你一个整数数组 nums ，数组中的元素 互不相同 。返回该数组所有可能的子集（幂集）。

解集 不能 包含重复的子集。你可以按 任意顺序 返回解集。

示例 1：
输入：nums = [1,2,3]
输出：[[],[1],[2],[1,2],[3],[1,3],[2,3],[1,2,3]]

示例 2：
输入：nums = [0]
输出：[[],[0]]


提示：
1 <= nums.length <= 10
-10 <= nums[i] <= 10
nums 中的所有元素 互不相同
```



```cpp
class Solution {
public:
    vector<vector<int>> ans;
    int n;
    vector<int> path;
    vector<bool> st;
    vector<vector<int>> subsets(vector<int>& nums) 
    {
        n = nums.size();
        st.resize(n);
        sort(nums.begin(), nums.end());
        dfs(nums, 0, 0);

        return ans;
    }

    void dfs(vector<int>& nums, int u, int start)
    {
        if(u == n)
        {
            ans.push_back(path);
            return;
        }
        
        if(path.empty())
            dfs(nums, u + 1, 0);

        for(int i = start; i < n; ++i)
            if(!st[i])
            {
                st[i] = true;
                path.push_back(nums[i]);
                dfs(nums, u + 1, i + 1);
                path.pop_back();
                st[i] = false;
            }
    }
};
```



```cpp
class Solution {
public:
    vector<vector<int>> ans;
    int n;
    vector<int> path;
    vector<bool> st;
    vector<vector<int>> subsets(vector<int>& nums) 
    {
        n = nums.size();
        st.resize(n);
        sort(nums.begin(), nums.end());
        dfs(nums, 0);

        return ans;
    }

    void dfs(vector<int>& nums, int u)
    {
        if(u == n)
        {
            ans.push_back(path);
            return;
        }
        
        dfs(nums, u + 1);

        path.push_back(nums[u]);
        dfs(nums, u + 1);
        path.pop_back();

    }
};
```
